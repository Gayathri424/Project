pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/lohithak28/WebProject.git'
            }
        }

        stage('Build WAR') {
            steps {
                bat "mvn clean package"
            }
        }

        stage('Deploy to Tomcat') {
            steps {
                bat "copy target\\*.war \"C:\\tomcat\\apache-tomcat-9.0.108\\webapps\\\" /Y"
            }
        }

        stage('Restart Tomcat') {
            steps {
                bat "\"C:\\tomcat\\apache-tomcat-9.0.108\\bin\\shutdown.bat\""
                bat "\"C:\\tomcat\\apache-tomcat-9.0.108\\bin\\startup.bat\""
            }
        }
    }
}

pipeline {
    agent any

    stages {
        stage('Checkout Code') {
            steps {
                git 'https://github.com/lohithak28/week9maven-1.git'
            }
        }

        stage('Build WAR') {
            steps {
                bat "mvn clean package"
            }
        }

        stage('Archive WAR') {
            steps {
                archiveArtifacts artifacts: 'target/*.jar', fingerprint: true
            }
        }
    }
}


//java + webhook
pipeline {
    agent any
    
    tools {
        maven 'MAVEN_HOME'
    }

    triggers {
        githubPush()   // GitHub Webhook Trigger
    }

    stages {

        stage('Clone Code') {
            steps {
                echo "Cloning Java Project..."
                git branch: 'main',
                    url: 'https://github.com/Gayathri424/SampleMavenJavaproject.git'
            }
        }

        stage('Build') {
            steps {
                echo "Running Maven Build..."
                bat 'mvn clean install'
            }
        }

        stage('Test') {
            steps {
                echo "Running Tests..."
                bat 'mvn test'
            }
        }

        stage('Package') {
            steps {
                echo "Packaging..."
                bat 'mvn package'
            }
        }

        stage('Success') {
            steps {
                echo "BUILD SUCCESS â€” Triggered by GitHub Webhook!"
            }
        }
    }
}
